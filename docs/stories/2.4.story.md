## Story

**As a** procurement manager,  
**I want to** upload required documents during qualification,  
**so that** we collect all necessary documentation before approval.

---

## Acceptance Criteria

1. Qualification workflow detail page displays document checklist section
2. Checklist displays all required documents with status indicators: Not uploaded (gray), Uploaded (blue), Approved (green), Rejected (red)
3. Each checklist item shows: Document name, Description, Required badge, Upload status, Upload button
4. Clicking "Upload" opens file picker and document metadata form
5. User can upload file and map it to checklist item (links existing supplier document OR uploads new document)
6. Progress bar shows upload progress for each document
7. After upload, checklist item status updates to "Uploaded (Pending Review)"
8. Document list shows all uploaded documents with: Filename, Uploaded by, Upload date, Status
9. Can upload multiple documents per checklist item if needed
10. Can remove uploaded document and re-upload (Admin/Procurement Manager only)
11. Checklist completion percentage displayed: "5 of 8 documents uploaded (63%)"
12. Cannot submit workflow for Stage 1 approval until all required documents are uploaded
13. Mobile-responsive with touch-optimized file upload interface

#### Workflow Document Upload Logic

**Two Upload Options:**

1. **Upload New Document** (AC 5 - uploads new document):
   - User selects file from device
   - File uploaded to Supabase Storage: `{tenantId}/{supplierId}/qualification-{workflowId}/{uuid}_{filename}`
   - Document record created in `documents` table with supplier_id from workflow
   - workflow_documents record updated: set document_id, status = "Uploaded"

2. **Link Existing Supplier Document** (AC 5 - links existing document):
   - User selects from dropdown of supplier's existing documents (from `documents` table)
   - No file upload needed
   - workflow_documents record updated: set document_id (existing), status = "Uploaded"
   - Reuses existing document in storage

**Critical Implementation Notes:**
- **Checklist Item ID Matching**: Each workflow_documents record has a `checklist_item_id` that references an item in the workflow's `snapshotted_checklist` JSONB array. Match by the item's `id` field (generated when workflow created).
- **Multiple Documents Per Item** (AC 9): Can upload multiple documents for same checklist item by creating multiple workflow_documents records with same checklist_item_id but different document_ids.
- **Remove Document** (AC 10): Set workflow_documents.document_id = NULL and status = "Pending". Do NOT delete from documents table or storage (audit trail).
- **Completion Calculation** (AC 11): Count required checklist items (where required=true) and count workflow_documents with status="Uploaded" or "Approved". Formula: `(uploadedCount / requiredCount) * 100`

**Implementation Pattern from Story 1.8:**
```typescript
// ElysiaJS file upload validation
import { Elysia, t } from "elysia";

export const uploadDocumentRoute = new Elysia({ prefix: "/workflows" })
  .use(authenticate)
  .post("/:workflowId/documents", async ({ params, body, user, set }: any) => {
    // Role check
    if (!user?.role || (user.role !== UserRole.PROCUREMENT_MANAGER && user.role !== UserRole.ADMIN)) {
      set.status = 403;
      return { success: false, error: { code: "FORBIDDEN", message: "Access denied" } };
    }
    
    // Verify workflow belongs to tenant
    const workflow = await db.query.qualificationWorkflows.findFirst({
      where: and(
        eq(qualificationWorkflows.id, params.workflowId),
        eq(qualificationWorkflows.tenantId, user.tenantId)
      ),
    });
    
    if (!workflow) {
      set.status = 404;
      return { success: false, error: { code: "NOT_FOUND", message: "Workflow not found" } };
    }
    
    // Option A: Upload new document
    if (body.file) {
      // Validate file
      const allowedTypes = ["application/pdf", "image/png", "image/jpeg", ...];
      if (!allowedTypes.includes(body.file.type)) {
        set.status = 400;
        return { success: false, error: { code: "INVALID_FILE_TYPE", message: "File type not supported" } };
      }
      
      if (body.file.size > 10 * 1024 * 1024) {
        set.status = 400;
        return { success: false, error: { code: "FILE_TOO_LARGE", message: "File size exceeds 10MB" } };
      }
      
      // Upload to Supabase Storage
      const uuid = crypto.randomUUID();
      const storagePath = `${user.tenantId}/${workflow.supplierId}/qualification-${workflow.id}/${uuid}_${body.file.name}`;
      
      const { data: uploadData, error: uploadError } = await supabaseAdmin.storage
        .from("supplier-documents")
        .upload(storagePath, body.file, {
          contentType: body.file.type,
          upsert: false,
        });
      
      if (uploadError) {
        set.status = 500;
        return { success: false, error: { code: "UPLOAD_FAILED", message: "File upload failed" } };
      }
      
      // Create document record
      const newDocument = await db.insert(documents).values({
        tenantId: user.tenantId,
        supplierId: workflow.supplierId,
        filename: body.file.name,
        documentType: body.documentType || "other",
        storagePath: storagePath,
        fileSize: body.file.size,
        mimeType: body.file.type,
        description: body.description || null,
        uploadedBy: user.id,
      }).returning();
      
      // Update workflow_documents record
      await db.update(workflowDocuments)
        .set({ documentId: newDocument[0].id, status: "Uploaded", updatedAt: new Date() })
        .where(and(
          eq(workflowDocuments.workflowId, workflow.id),
          eq(workflowDocuments.checklistItemId, body.checklistItemId)
        ));
      
      set.status = 201;
      return { success: true, data: { document: newDocument[0] } };
    }
    
    // Option B: Link existing document
    if (body.documentId) {
      // Verify document belongs to same supplier
      const existingDoc = await db.query.documents.findFirst({
        where: and(
          eq(documents.id, body.documentId),
          eq(documents.supplierId, workflow.supplierId),
          eq(documents.tenantId, user.tenantId)
        ),
      });
      
      if (!existingDoc) {
        set.status = 404;
        return { success: false, error: { code: "DOCUMENT_NOT_FOUND", message: "Document not found" } };
      }
      
      // Update workflow_documents record
      await db.update(workflowDocuments)
        .set({ documentId: body.documentId, status: "Uploaded", updatedAt: new Date() })
        .where(and(
          eq(workflowDocuments.workflowId, workflow.id),
          eq(workflowDocuments.checklistItemId, body.checklistItemId)
        ));
      
      return { success: true, data: { document: existingDoc } };
    }
    
    set.status = 400;
    return { success: false, error: { code: "INVALID_REQUEST", message: "Must provide file or documentId" } };
  }, {
    body: t.Object({
      checklistItemId: t.String({ format: "uuid" }),
      file: t.Optional(t.File({ maxSize: 10485760 })), // 10MB
      documentId: t.Optional(t.String({ format: "uuid" })),
      documentType: t.Optional(t.String()),
      description: t.Optional(t.String()),
    }),
  });
```

export async function loader(args: LoaderFunctionArgs) {
  // 1. Authentication
  const { session } = await requireAuth(args);
  
  // 2. Get params
  const { id } = args.params;
  if (!id) throw new Response("Workflow ID required", { status: 400 });
  
  // 3. Create API client
  const token = session?.access_token;
  if (!token) throw new Response("Unauthorized", { status: 401 });
  const client = createEdenTreatyClient(token);
  
  // 4. Fetch data in parallel
  const [workflowResponse, documentsResponse, supplierDocsResponse] = await Promise.all([
    client.api.workflows[id].get(),
    client.api.workflows[id].documents.get(),
    // Fetch supplier's existing documents for "Link Existing" option
    // (assuming workflow includes supplier_id in response)
  ]);
  
  if (workflowResponse.error) {
    const status = workflowResponse.status || 500;
    if (status === 404) {
      throw new Response("Workflow not found", { status: 404 });
    }
    throw new Response("Failed to load workflow", { status });
  }
  
  // 5. Return all data
  return json({
    workflow: workflowResponse.data.workflow,
    workflowDocuments: documentsResponse.data?.workflowDocuments || [],
    supplierDocuments: supplierDocsResponse.data?.documents || [],
    token,
  });
}

export function shouldRevalidate({ currentUrl, nextUrl, defaultShouldRevalidate }) {
  // Don't revalidate on search param changes
  if (currentUrl.pathname === nextUrl.pathname) {
    if (currentUrl.searchParams.toString() !== nextUrl.searchParams.toString()) {
      return false;
    }
  }
  return defaultShouldRevalidate;
}

export default function WorkflowDetail() {
  const { workflow, workflowDocuments, supplierDocuments, token } = useLoaderData<typeof loader>();
  
  return (
    <div>
      <WorkflowDetailPage 
        workflow={workflow} 
        workflowDocuments={workflowDocuments}
        supplierDocuments={supplierDocuments}
        token={token} 
      />
    </div>
  );
}

import { Dialog, DialogContent, DialogHeader, DialogTitle } from "~/components/ui/dialog";
import { Tabs, TabsContent, TabsList, TabsTrigger } from "~/components/ui/tabs";
import { useRevalidator } from "@remix-run/react";

export function UploadWorkflowDocumentModal({ open, onOpenChange, workflowId, checklistItem, supplierDocuments, token }) {
  const [uploadOption, setUploadOption] = useState<"new" | "existing">("new");
  const [isUploading, setIsUploading] = useState(false);
  const revalidator = useRevalidator();
  const { toast } = useToast();

  const handleUploadNew = async (file: File) => {
    setIsUploading(true);
    try {
      const formData = new FormData();
      formData.append("file", file);
      formData.append("checklistItemId", checklistItem.id);
      
      const client = createClientEdenTreatyClient(token);
      const response = await client.api.workflows[workflowId].documents.post(formData);
      
      if (response?.error) {
        throw new Error(response.error.message || "Upload failed");
      }
      
      toast({
        title: "Document uploaded successfully",
        variant: "success",
      });
      
      revalidator.revalidate();
      onOpenChange(false);
    } catch (error: any) {
      toast({
        title: "Upload failed",
        description: error.message || "Failed to upload document",
        variant: "destructive",
      });
    } finally {
      setIsUploading(false);
    }
  };

  const handleLinkExisting = async (documentId: string) => {
    setIsUploading(true);
    try {
      const client = createClientEdenTreatyClient(token);
      const response = await client.api.workflows[workflowId].documents.post({
        checklistItemId: checklistItem.id,
        documentId: documentId,
      });
      
      if (response?.error) {
        throw new Error(response.error.message || "Failed to link document");
      }
      
      toast({
        title: "Document linked successfully",
        variant: "success",
      });
      
      revalidator.revalidate();
      onOpenChange(false);
    } catch (error: any) {
      toast({
        title: "Link failed",
        description: error.message || "Failed to link document",
        variant: "destructive",
      });
    } finally {
      setIsUploading(false);
    }
  };

  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent className="max-w-2xl">
        <DialogHeader>
          <DialogTitle>Upload Document: {checklistItem.name}</DialogTitle>
        </DialogHeader>
        
        <Tabs value={uploadOption} onValueChange={(v) => setUploadOption(v as "new" | "existing")}>
          <TabsList>
            <TabsTrigger value="new">Upload New</TabsTrigger>
            <TabsTrigger value="existing">Link Existing</TabsTrigger>
          </TabsList>
          
          <TabsContent value="new">
            {/* File picker, validation, progress bar */}
          </TabsContent>
          
          <TabsContent value="existing">
            {/* Dropdown of supplier documents */}
          </TabsContent>
        </Tabs>
      </DialogContent>
    </Dialog>
  );
}
